# LLM Interview Simulator - Python Dependencies
# =============================================
# This file lists all the Python packages needed to run our FastAPI backend.
# Install these with: pip install -r requirements.txt
#
# Why each package is needed:
# - fastapi: The main web framework for building our API
# - uvicorn: ASGI server to run our FastAPI application
# - python-dotenv: Load environment variables from .env files
# - pydantic: Data validation and serialization for our API models
# - python-multipart: Handle file uploads and form data
# - gunicorn: Production WSGI server (alternative to uvicorn)

# Core FastAPI framework and server
fastapi==0.104.1              # Modern, fast web framework for building APIs
uvicorn[standard]==0.24.0      # ASGI server with additional features (websockets, etc.)

# Environment and configuration
python-dotenv==1.0.0          # Load environment variables from .env files

# Data validation and serialization
pydantic==2.5.0               # Data validation using Python type annotations

# File handling and forms
python-multipart==0.0.6       # Handle multipart/form-data (file uploads, forms)

# Production server (alternative to uvicorn)
gunicorn==21.2.0              # WSGI HTTP Server for production deployment
